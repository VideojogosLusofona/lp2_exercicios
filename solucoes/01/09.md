```cs
            //Escreve ao User que dados esta a introduzir
            Console.Write("Insert Number to be checked:");

            //User input para o numero a ser procurado
            value = Convert.ToInt32(Console.ReadLine());
            int[,] arr = new int[x, y];
            
            //Cria uma instancia da Classe Checker
            Checker check1 = new Checker();
            
            //Verifica se o numero se repete as vezes necessárias e

            else Console.WriteLine("It failed");
        }
    }
}
```

Class Checker
```cs
namespace Ex1_09
{
    public class Checker
    {
        public static bool Check(int[,] x, int y)
        {
            //Declaração de Variáveis
            int counter = 0;

            //Ciclos que verificam em ambas as linhas e colunas se o numero 
            //pedido se encontra repetido 4 ou mais vezes.
            for (int i = 0; i < x.GetLength(0); i++)
            {
                for(int j = 0; j < x.GetLength(1); j++)
                {
                    if (i == y || i == j) counter++;

                }
            }

            //Retorna o resultado da operação
            if(counter >= 4)
            {
                return true;
            }
            return false;        
        }
    }
}
```
